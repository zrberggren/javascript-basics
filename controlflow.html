If-Else Statements


// Example of If-Else Statements
let age = 18;
if (age >= 18) {
  console.log("You are an adult.");
} else {
  console.log("You are a minor.");
}

Switch Statements


// Example of Switch Statements
let day = 3;
switch (day) {
  case 1:
    console.log("Monday");
    break;
  case 2:
    console.log("Tuesday");
    break;
  case 3:
    console.log("Wednesday");
    break;
  default:
    console.log("Another day");
}

Loops


// Example of For Loop
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// Example of While Loop
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}
