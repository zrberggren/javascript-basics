JavaScript Operators

Operators are used to perform operations on variables and values.


 let name = "John"; // String Literal
 let Age = 31; // Number Literal
 let isCool = true; // Boolean Literal
 let firstName = undefined; // Undefined
 let selectedColor = Null; // Null

let person = {
  name: "John",
  age: 31,
}; // Object Literal

// Dot Notation (Best Practice)
person.name = "John";

// Bracket Notation
let selection = "name";
person[selection] = "Mary";

console.log(person.name);

let selectedColors = ["red", "blue"]; // Array Literal
selectedColors[2] = 1; // Add to Array
console.log(selectedColors.length); // Length of Array

// Perform a task
function greet(name, lastName) {
  // name is a parameter

  console.log("Hello " + name + " " + lastName);
}

greet("John", "Smith"); // John is an argument

// Calculate a value
function square(number) {
  return number * number;
}

console.log(square(2));

Assignment Operators

Assignment operators are used to assign values to variables.


// Example of Assignment Operators
let x = 10;
x += 5; // x = x + 5
console.log(x); // 15

Arithmetic Operators

Arithmetic operators are used to perform arithmetic on numbers.


// Example of Arithmetic Operators
let a = 5;
let b = 2;
let sum = a + b;
console.log(sum); // 7

Comparison Operators

Comparison operators are used to compare two values.


// Example of Comparison Operators
console.log(5 > 3); // true
console.log(5 < 3); // false

Equality Operators

Equality operators are used to compare two values for equality.


// Example of Equality Operators
console.log(1 == '1'); // true
console.log(1 === '1'); // false

Ternary Operators

Ternary operators are used to assign a value based on a condition. 


// Example of Ternary Operators
let age = 18;
let canVote = (age >= 18) ? 'Yes' : 'No';
console.log(canVote); // Yes

Logical Operators

Logical operators are used to combine multiple conditions.


// Example of Logical Operators
console.log(true && false); // false
console.log(true || false); // true
console.log(!true); // false

Bitwise Operators

Bitwise operators are used to perform bit-level operations.


// Example of Bitwise Operators
console.log(5 & 1); // 1
console.log(5 | 1); // 5
console.log(~5); // -6
console.log(5 ^ 1); // 4
console.log(5 << 1); // 10
console.log(5 >> 1); // 2

Operator Precedence

Operator precedence determines the order in which operators are evaluated. 


// Example of Operator Precedence
let result = 3 + 4 * 5; // 23
console.log(result);
result = (3 + 4) * 5; // 35
console.log(result);
